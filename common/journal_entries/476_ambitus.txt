	immediate = {

		ig:ig_devout = {

		ig:ig_industrialists = {

			save_scope_as = crisis_of_illegitimacy_ig_armed_forces
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = crisis_of_illegitimacy_ig_devout_txt
				
				trigger = {
					NOT = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_devout }
							ig_clout > scope:crisis_of_illegitimacy_ig_devout.ig_clout
						}
					}
				}
			}
			triggered_desc = {
				desc = crisis_of_illegitimacy_ig_industrialists_txt
				
				trigger = {
					NOT = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_industrialists }
							ig_clout > scope:crisis_of_illegitimacy_ig_industrialists.ig_clout
						}
					}
				}
			}
			triggered_desc = {
				desc = crisis_of_illegitimacy_ig_intelligentsia_txt
				
				trigger = {
					NOT = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_intelligentsia }
							ig_clout > scope:crisis_of_illegitimacy_ig_intelligentsia.ig_clout
						}
					}
				}
			}
			triggered_desc = {
				desc = crisis_of_illegitimacy_ig_landowners_txt
				
				trigger = {
					NOT = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_landowners }

								NOR = {
									is_ruler = yes
									is_heir = yes

							}
							ig_clout > scope:crisis_of_illegitimacy_ig_landowners.ig_clout
						}
					}
				}
			}
			triggered_desc = {
				desc = status_dynasty_cementing_fra
				
				trigger = {
					NOT = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_petty_bourgeoisie }
							leader = {
								NOR = {
									is_ruler = yes
									is_heir = yes
								}
							}
							this.ig_clout > ig:ig_petty_bourgeoisie.ig_clout
						}
					}
				}
			}
			triggered_desc = {
				desc = status_dynasty_cementing_fra
				
				trigger = {

					NOT = {

	}

	complete = {
		NOT = {
			any_interest_group = {
				is_powerful = yes
				leader = {
					is_ruler = no
					has_role = general
					OR = {
						has_trait = ambitious
						has_trait = reckless
					}
				}
				leader.popularity > owner.ruler.popularity
			}
		}
		custom_tooltip = {
			text = caudillo_complete
			root.var:caudillo_var <= 0
		}
	}

	on_complete = {
		trigger_event = { id = fall_of_rome_events.2 popup = yes }
		show_as_tooltip = {
			ig:ig_armed_forces = {
				remove_ideology = ideology_caudillismo
			}
			remove_modifier = age_of_caudillos_modifier
		}
	}

	fail = {
		custom_tooltip = {
			text = je_caudillo_goal_tt
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
	}

	on_fail = {
		trigger_event = { id = fall_of_rome_events.3 popup = yes } # No More Games
		custom_tooltip = crisis_of_illegitimacy_rome_dies
	}

	on_weekly_pulse = {
		effect = {
            calculate_ambitus_progress = yes
        }
	}

	on_monthly_pulse = {
		effect = {
			change_variable = { name = caudillo_var add = var:caudillo_next }
		}

		random_events = {
			100 = 0 
			20 = fall_of_rome_events.11 # The Sword of Ambition
		}
	}

	on_yearly_pulse = {
		random_events = {
			70 = 0
			30 = fall_of_rome_events.4 # Conflict of the Urban Orders
			30 = fall_of_rome_events.7 # Language Reform?
			30 = fall_of_rome_events.9 # Bagaudae Disturb State
		}
	}

	current_value = {

	}

	goal_add_value = {
		add = 1200
	}

	transferable = yes
	should_be_pinned_by_default = yes
