refused_to_do_what_was_expected_effect = {
	if = {
		limit = {
			NOR = {
				has_modifier = refused_to_do_what_was_expected_1
				has_modifier = refused_to_do_what_was_expected_2
				has_modifier = refused_to_do_what_was_expected_3
				has_modifier = refused_to_do_what_was_expected_4
				has_modifier = refused_to_do_what_was_expected_5
			}
		}
		add_modifier = {
			name = refused_to_do_what_was_expected_1
			months = short_modifier_time
		}
	}
	else_if = {
		limit = { has_modifier = refused_to_do_what_was_expected_1 }
		hidden_effect = { remove_modifier = refused_to_do_what_was_expected_1 }
		add_modifier = {
			name = refused_to_do_what_was_expected_2
			months = short_modifier_time
		}
	}
	else_if = {
		limit = { has_modifier = refused_to_do_what_was_expected_2 }
		hidden_effect = { remove_modifier = refused_to_do_what_was_expected_2 }
		add_modifier = {
			name = refused_to_do_what_was_expected_3
			months = short_modifier_time
		}
	}
	else_if = {
		limit = { has_modifier = refused_to_do_what_was_expected_3 }
		hidden_effect = { remove_modifier = refused_to_do_what_was_expected_3 }
		add_modifier = {
			name = refused_to_do_what_was_expected_4
			months = short_modifier_time
		}
	}
	else_if = {
		limit = { has_modifier = refused_to_do_what_was_expected_4 }
		hidden_effect = { remove_modifier = refused_to_do_what_was_expected_4 }
		add_modifier = {
			name = refused_to_do_what_was_expected_5
			months = short_modifier_time
		}
	}
	else_if = {
		limit = { has_modifier = refused_to_do_what_was_expected_5 }
		hidden_effect = { remove_modifier = refused_to_do_what_was_expected_5 }
		custom_tooltip = ruler_extremely_unpopular
		add_modifier = {
			name = refused_to_do_what_was_expected_5
			months = short_modifier_time
		}
	}
}

ambitus_caudillo_progress = {
	set_variable = { name = caudillo_next value = 0 }
	if = {
		limit = {
			any_interest_group = {
				is_powerful = yes
				leader = {
					is_ruler = no
					OR = {
						has_trait = ambitious
						has_trait = reckless
					}
				}
			}
		}
		change_variable = {
			name = caudillo_next
			add = 1
		}
	}
	if = {
		limit = {
			any_interest_group = {
				is_in_government = yes
				leader = {
					is_ruler = no
					OR = {
						has_trait = ambitious
						has_trait = reckless
					}
				}
			}
		}
		change_variable = {
			name = caudillo_next
			add = 1
		}
	}
	if = {
		limit = {
			any_interest_group = {
				is_powerful = yes
				leader.popularity > owner.ruler.popularity
			}
		}
		change_variable = {
			name = caudillo_next
			add = 1
		}
	}
	if = {
		limit = {
			NOT = {
				any_interest_group = {
					is_powerful = yes
					leader.popularity > owner.ruler.popularity
				}
			}
		}
		change_variable = {
			name = caudillo_next
			add = -1
		}
	}
	if = {
		limit = {
			NOT = {
				any_interest_group = {
					is_marginal = no
					leader = {
						is_ruler = no
						OR = {
							has_trait = ambitious
							has_trait = reckless
						}
					}
				}
			}
		}
		change_variable = {
			name = caudillo_next
			add = -1
		}
	}
	if = {
		limit = {
			any_scope_character = {
				OR = {
					has_role = general
					has_role = admiral
				}
				popularity >= 50 # Loved
			}
		}
		change_variable = {
			name = caudillo_next
			add = 1
		}
	}
	if = {
		limit = {
			country_has_voting_franchise = yes
		}
		change_variable = {
			name = caudillo_next
			add = -1
		}
	}
	if = {
		limit = {
			any_scope_character = {
				is_ruler = no
				is_heir = no
				is_agitator = no
				is_politician = yes
				has_ideology = ideology:ideology_dominate
			}
		}
		change_variable = {
			name = caudillo_next
			add = 1
		}
	}
	if = {
		limit = {
			OR = {
				has_law = law_type:law_census_voting
				has_law = law_type:law_universal_suffrage
			}
		}
		change_variable = {
			name = caudillo_next
			add = -1
		}
	}
}
